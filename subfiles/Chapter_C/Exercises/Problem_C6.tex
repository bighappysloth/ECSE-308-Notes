\subsubsection{Problem C6}
\begin{wts}
    Find the Hamming distance between the following codewords: A = [000000], B = [101010], C = [000111]
\end{wts}
\begin{proof}
This is trivial, $d(A,B)=\wt(B)=3$, $d(A,C)=\wt(C)=3$, and $d(B,C) = 4$. The minimum distance is $3$. 
\end{proof}
